```typescript
// From src/types/index.ts:
export type SourceType = 'file' | 'url' | 'codebase' | 'text';
export type SourceStatus = 'pending' | 'indexing' | 'indexed' | 'error' | 'stale';

export interface KnowledgeSource {
  id: string;
  name: string;
  type: SourceType;
  sourcePath: string | null;
  fingerprint: string | null;
  status: SourceStatus;
  chunkCount: number;
  documentCount: number;
  lastIndexedAt: string | null;
  createdAt: string;
  updatedAt: string;
  errorMessage: string | null;
  tag: string;
}

export interface RetrievedChunk {
  chunkId: string;
  sourceId: string;
  sourceName: string;
  originPath: string;
  content: string;
  score: number;
  tag: string;
}
```

<!-- API endpoints to call from client components: -->
```
GET  /api/knowledge/sources          → { sources: KnowledgeSource[] }
POST /api/knowledge/sources          → KnowledgeSource (body: { name, type, sourcePath })
GET  /api/knowledge/sources/[id]     → KnowledgeSource
DELETE /api/knowledge/sources/[id]   → { deleted: true }
POST /api/knowledge/sources/[id]/index    → { message, result }
POST /api/knowledge/sources/[id]/reindex  → { reindexed, reason, result? }
POST /api/knowledge/search           → { chunks: RetrievedChunk[], count }
```
