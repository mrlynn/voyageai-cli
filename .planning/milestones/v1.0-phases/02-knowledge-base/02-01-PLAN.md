export type SourceType = 'file' | 'url' | 'codebase' | 'text';
export type SourceStatus = 'pending' | 'indexing' | 'indexed' | 'error' | 'stale';

export interface KnowledgeSource {
  id: string;
  name: string;
  type: SourceType;
  // For file/codebase sources: absolute path. For url: the URL. For text: null.
  sourcePath: string | null;
  // Fingerprint for change detection: file hash (SHA-256) or web ETag
  fingerprint: string | null;
  status: SourceStatus;
  chunkCount: number;
  documentCount: number;
  // ISO string of last successful indexing
  lastIndexedAt: string | null;
  createdAt: string;
  updatedAt: string;
  // Error message if status === 'error'
  errorMessage: string | null;
  // Tag for metadata filtering (e.g. 'docs', 'codebase', 'web')
  tag: string;
}

export interface IndexedChunk {
  id: string;
  sourceId: string;
  // Human-readable source name for attribution
  sourceName: string;
  content: string;

  embedding: number[];
  // Position within source document
  chunkIndex: number;
  totalChunks: number;
  // Original file path or URL within source
  originPath: string;
  metadata: Record<string, unknown>;
  createdAt: string;
}

export interface SourceVersion {
  id: string;
  sourceId: string;

Create a MongoDB data access module. Database name: `vai_dashboard`. Collections: `knowledge_sources` (KnowledgeSource documents) and `knowledge_chunks` (IndexedChunk documents with vector embeddings).

```typescript
// src/lib/db/knowledge.ts
import { MongoClient, Collection, ObjectId } from 'mongodb';
import type { KnowledgeSource, IndexedChunk, SourceVersion } from '@/types';

// Cache MongoDB client across hot reloads in Next.js dev mode
declare global {
  var _mongoKnowledgeClient: MongoClient | undefined;
}

const DB_NAME = 'vai_dashboard';
const SOURCES_COLLECTION = 'knowledge_sources';
const CHUNKS_COLLECTION = 'knowledge_chunks';
const VERSIONS_COLLECTION = 'knowledge_versions';

function requireMongoUri(): string {
  const uri = process.env.MONGODB_URI;
  if (!uri) {
    throw new Error('MONGODB_URI environment variable is not set. Add it to .env.local');
  }
  return uri;
}

async function getClient(): Promise<MongoClient> {
  if (global._mongoKnowledgeClient) {
    return global._mongoKnowledgeClient;
  }
  const client = new MongoClient(requireMongoUri());
  await client.connect();
  if (process.env.NODE_ENV !== 'production') {
    global._mongoKnowledgeClient = client;
  }
  return client;
}

export async function getKnowledgeCollection(): Promise<{ client: MongoClient; collection: Collection<KnowledgeSource> }> {
  const client = await getClient();
  const collection = client.db(DB_NAME).collection<KnowledgeSource>(SOURCES_COLLECTION);
  return { client, collection };
}

export async function getChunksCollection(): Promise<{ client: MongoClient; collection: Collection<IndexedChunk> }> {
  const client = await getClient();
  const collection = client.db(DB_NAME).collection<IndexedChunk>(CHUNKS_COLLECTION);
  return { client, collection };
}

export async function getVersionsCollection(): Promise<{ client: MongoClient; collection: Collection<SourceVersion> }> {
  const client = await getClient();
  const collection = client.db(DB_NAME).collection<SourceVersion>(VERSIONS_COLLECTION);
  return { client, collection };
}

// Tag inference based on source type
function inferTag(type: string, sourcePath: string | null): string {
  if (type === 'url') return 'web';
  if (type === 'codebase') return 'codebase';
  if (type === 'text') return 'pasted';
  // file — use path extension hints
  return 'docs';
}

export function buildSourceDocument(
  input: Pick<KnowledgeSource, 'name' | 'type' | 'sourcePath'>
): KnowledgeSource {
  const now = new Date().toISOString();
  return {
    id: crypto.randomUUID(),
    name: input.name,
    type: input.type,
    sourcePath: input.sourcePath,
    fingerprint: null,
    status: 'pending',
    chunkCount: 0,
    documentCount: 0,
    lastIndexedAt: null,
    createdAt: now,
    updatedAt: now,
    errorMessage: null,
    tag: inferTag(input.type, input.sourcePath),
  };
}
```
