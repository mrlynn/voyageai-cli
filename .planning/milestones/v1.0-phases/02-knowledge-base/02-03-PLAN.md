```typescript
import { generateEmbeddings } from '@/lib/ingestion/embed';
import { getChunksCollection, getKnowledgeCollection } from '@/lib/db/knowledge';
import type { RetrievedChunk, SearchRequest } from '@/types';

export async function vectorSearch(request: SearchRequest): Promise<RetrievedChunk[]> {
  const { query, topK = 8, minScore = 0.65, sourceIds, tags } = request;

  // 1. Embed the query (input_type should be 'query' not 'document' for search queries)
  //    Note: generateEmbeddings in embed.ts uses 'document' input_type.
  //    For search, we need 'query' input_type. Make a direct fetch call here OR
  //    accept the slight quality loss and reuse generateEmbeddings (acceptable for v1).
  //    Use direct fetch with input_type: 'query' for correctness:
  const apiKey = process.env.VOYAGE_API_KEY;
  if (!apiKey) throw new Error('VOYAGE_API_KEY is not set');

  const base = process.env.VOYAGE_API_BASE || 'https://api.voyageai.com/v1';
  const model = process.env.VOYAGE_MODEL || 'voyage-3';

  const embedRes = await fetch(`${base}/embeddings`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` },
    body: JSON.stringify({ input: [query], model, input_type: 'query' }),
  });
  if (!embedRes.ok) {
    const err = await embedRes.text();
    throw new Error(`Voyage AI embedding error: ${embedRes.status} ${err}`);
  }
  const embedData = await embedRes.json() as { data: Array<{ embedding: number[] }> };
  const queryVector = embedData.data[0].embedding;

  // 2. Build $vectorSearch pipeline
  const numCandidates = Math.min(topK * 15, 1000);

  // Build pre-filter for optional source/tag filtering
  // Note: $vectorSearch filter uses pre-filter (applied before ANN search)
  // Only simple equality/in filters are supported in pre-filter
  const vectorSearchStage: Record<string, unknown> = {
    index: 'vector_index',
    path: 'embedding',
    queryVector,
    numCandidates,
    limit: topK,
  };

  if (sourceIds && sourceIds.length > 0) {
    vectorSearchStage.filter = { sourceId: { $in: sourceIds } };
  }
  // Note: tag filtering requires a join to knowledge_sources — do it post-search instead

  const pipeline = [
    { $vectorSearch: vectorSearchStage },
    { $addFields: { score: { $meta: 'vectorSearchScore' } } },
    { $match: { score: { $gte: minScore } } },
  ];

  const { collection: chunksCollection } = await getChunksCollection();
  const rawResults = await chunksCollection.aggregate(pipeline).toArray();

  // 3. Post-filter by tag if requested (requires knowledge_sources lookup)
  let results = rawResults;
  if (tags && tags.length > 0) {
    const { collection: sourcesCollection } = await getKnowledgeCollection();
    const sourceIdsWithTags = await sourcesCollection
      .find({ tag: { $in: tags } }, { projection: { id: 1 } })
      .toArray();
    const taggedSourceIds = new Set(sourceIdsWithTags.map(s => s.id));
    results = rawResults.filter(r => taggedSourceIds.has(r.sourceId));
  }

  // 4. Get source metadata for tag enrichment on results
  const uniqueSourceIds = [...new Set(results.map(r => r.sourceId))];
  const { collection: sourcesCollection } = await getKnowledgeCollection();
  const sources = await sourcesCollection
    .find({ id: { $in: uniqueSourceIds } }, { projection: { id: 1, tag: 1 } })
    .toArray();
  const sourceTagMap = new Map(sources.map(s => [s.id, s.tag]));

  // 5. Map to RetrievedChunk
  return results.map(r => ({
    chunkId: r.id,
    sourceId: r.sourceId,
    sourceName: r.sourceName,
    originPath: r.originPath,
    content: r.content,
    score: r.score as number,
    tag: sourceTagMap.get(r.sourceId) || 'unknown',
  }));
}
```

**Step 3 — Create src/lib/retrieval/inject.ts:**

```typescript
import { vectorSearch } from './search';
import type { SearchRequest, RetrievedChunk } from '@/types';

// Default configuration for RAG injection
const DEFAULT_TOP_K = 8;
const DEFAULT_MIN_SCORE = 0.65;
const MAX_CONTEXT_CHARS = 8000; // Prevent context window overflow in OpenAI prompts

export async function retrieveContext(
  query: string,
  options?: Partial<SearchRequest>
): Promise<string[]> {
  // Returns string[] of chunk contents ready for injection into GenerationRequest.knowledgeContext
  const chunks = await vectorSearch({
    query,
    topK: options?.topK ?? DEFAULT_TOP_K,
    minScore: options?.minScore ?? DEFAULT_MIN_SCORE,
    sourceIds: options?.sourceIds,
    tags: options?.tags,
  });

  // Truncate to prevent context overflow: accumulate chunks until MAX_CONTEXT_CHARS
  const contextStrings: string[] = [];
  let totalChars = 0;

  for (const chunk of chunks) {
    // Format with source attribution for transparency in prompts
    const formatted = `[Source: ${chunk.sourceName} | ${chunk.originPath}]\n${chunk.content}`;
    if (totalChars + formatted.length > MAX_CONTEXT_CHARS) break;
    contextStrings.push(formatted);
    totalChars += formatted.length;
  }

  return contextStrings;
}

export async function retrieveContextWithMetadata(
  query: string,
  options?: Partial<SearchRequest>
): Promise<RetrievedChunk[]> {
  // Returns full RetrievedChunk objects (for test panel UI in Plan 02-04)
  return vectorSearch({
    query,
    topK: options?.topK ?? DEFAULT_TOP_K,
    minScore: options?.minScore ?? DEFAULT_MIN_SCORE,
    sourceIds: options?.sourceIds,
    tags: options?.tags,
  });
}
```

```typescript
// POST /api/knowledge/search
// Body: SearchRequest { query, topK?, minScore?, sourceIds?, tags? }
// Response 200: { chunks: RetrievedChunk[], count: number, queryEmbedded: true }
// Response 400: { error: 'query is required' }
// Response 503: { error: 'No knowledge sources indexed yet' } — caught from vectorSearch if no vector index
// Response 500: { error: string }

import { NextResponse } from 'next/server';
import { retrieveContextWithMetadata } from '@/lib/retrieval/inject';
import type { SearchRequest } from '@/types';

export async function POST(request: Request) {
  try {
    const body = await request.json() as Partial<SearchRequest>;

    if (!body.query || typeof body.query !== 'string' || body.query.trim() === '') {
      return NextResponse.json({ error: 'query is required' }, { status: 400 });
    }

    // Validate optional fields
    const searchRequest: SearchRequest = {
      query: body.query.trim(),
      topK: typeof body.topK === 'number' ? Math.min(body.topK, 20) : 8,
      minScore: typeof body.minScore === 'number' ? body.minScore : 0.65,
      sourceIds: Array.isArray(body.sourceIds) ? body.sourceIds : undefined,
      tags: Array.isArray(body.tags) ? body.tags : undefined,
    };

    const chunks = await retrieveContextWithMetadata(searchRequest.query, searchRequest);

    return NextResponse.json({
      chunks,
      count: chunks.length,
      queryEmbedded: true,
    });
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Unknown error';
    // Detect "no index" error from MongoDB — common on first use
    if (message.includes('index') && message.includes('not found')) {
      return NextResponse.json(
        { error: 'No vector search index found. Index a source first to create the index.' },
        { status: 503 }
      );
    }
    return NextResponse.json({ error: message }, { status: 500 });
  }
}
```
