```typescript
// src/lib/generation/index.ts
import { retrieveContext } from '@/lib/retrieval/inject';
import { buildContentPrompt } from '@/lib/generation/prompts';
import { getOpenAIClient, DEFAULT_MODEL } from '@/lib/openai';
import type { GenerationRequest, GenerationResponse, ContentDraft } from '@/types';

export interface GenerateOptions {
  contentType: GenerationRequest['contentType'];
  topic: string;
  platform?: GenerationRequest['platform'];
  additionalInstructions?: string;
  tags?: string[];      // optional: filter knowledge sources by tag
  sourceIds?: string[]; // optional: filter to specific sources
}

export async function generateWithContext(options: GenerateOptions): Promise<GenerationResponse> {
  // 1. Retrieve knowledge context (with retry + fallback)
  let knowledgeContext: string[] = [];
  try {
    knowledgeContext = await retrieveContext(options.topic, {
      tags: options.tags,
      sourceIds: options.sourceIds,
    });
  } catch (err) {
    // Retry once
    try {
      knowledgeContext = await retrieveContext(options.topic, {
        tags: options.tags,
        sourceIds: options.sourceIds,
      });
    } catch {
      // Proceed without context — graceful degradation
      console.warn('Knowledge retrieval failed, generating without context:', err);
    }
  }

  // 2. Build prompts
  const request: GenerationRequest = {
    contentType: options.contentType,
    topic: options.topic,
    platform: options.platform,
    knowledgeContext,
    additionalInstructions: options.additionalInstructions,
  };
  const { system, user } = buildContentPrompt(request);

  // 3. Call OpenAI
  const client = getOpenAIClient();
  const completion = await client.chat.completions.create({
    model: DEFAULT_MODEL,
    messages: [
      { role: 'system', content: system },
      { role: 'user', content: user },
    ],
    temperature: 0.7,
  });

  const content = completion.choices[0]?.message?.content;
  if (!content) {
    throw new Error('OpenAI returned an empty response');
  }

  // 4. Build response
  const draft: ContentDraft = {
    id: crypto.randomUUID(),
    type: options.contentType,
    title: options.topic,
    body: content,
    platform: options.platform,
    status: 'draft',
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
  };

  return {
    draft,
    tokensUsed: completion.usage?.total_tokens ?? 0,
    model: completion.model,
  };
}
```
