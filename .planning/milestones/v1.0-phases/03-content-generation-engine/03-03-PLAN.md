```typescript
// src/app/api/generate/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { generateWithContext } from '@/lib/generation';
import type { ContentType } from '@/types';

const VALID_CONTENT_TYPES: ContentType[] = ['blog-post', 'social-post', 'code-example', 'video-script'];

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();

    // Validate contentType
    if (!body.contentType || !VALID_CONTENT_TYPES.includes(body.contentType)) {
      return NextResponse.json(
        { error: `Invalid contentType. Must be one of: ${VALID_CONTENT_TYPES.join(', ')}` },
        { status: 400 }
      );
    }

    // Validate topic
    if (!body.topic || typeof body.topic !== 'string' || body.topic.trim() === '') {
      return NextResponse.json(
        { error: 'topic is required and must be a non-empty string' },
        { status: 400 }
      );
    }

    const result = await generateWithContext({
      contentType: body.contentType,
      topic: body.topic.trim(),
      platform: body.platform,
      additionalInstructions: body.additionalInstructions,
      tags: body.tags,
      sourceIds: body.sourceIds,
    });

    return NextResponse.json(result, { status: 201 });
  } catch (err) {
    console.error('Generation error:', err);
    return NextResponse.json(
      { error: err instanceof Error ? err.message : 'Content generation failed' },
      { status: 500 }
    );
  }
}
```
