    ```typescript
    import OpenAI from 'openai';
    import type { GenerationRequest, GenerationResponse, ContentDraft } from '@/types';

    let clientInstance: OpenAI | null = null;

    export function getOpenAIClient(): OpenAI {
      if (!process.env.OPENAI_API_KEY) {
        throw new Error('OPENAI_API_KEY environment variable is not set');
      }
      if (!clientInstance) {
        clientInstance = new OpenAI({
          apiKey: process.env.OPENAI_API_KEY,
        });
      }
      return clientInstance;
    }

    // Reset for testing
    export function _resetClientForTesting(): void {
      clientInstance = null;
    }

    export const DEFAULT_MODEL = process.env.OPENAI_MODEL || 'gpt-4o';

    export async function generateContent(request: GenerationRequest): Promise<GenerationResponse> {
      const client = getOpenAIClient();

      const systemPrompt = buildSystemPrompt(request);
      const userPrompt = buildUserPrompt(request);

      const completion = await client.chat.completions.create({
        model: DEFAULT_MODEL,
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userPrompt },
        ],
        temperature: 0.7,
      });

      const content = completion.choices[0]?.message?.content;
      if (!content) {
        throw new Error('OpenAI returned an empty response');
      }

      const draft: ContentDraft = {
        id: crypto.randomUUID(),
        type: request.contentType,
        title: request.topic,
        body: content,
        platform: request.platform,
        status: 'draft',
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
      };

      return {
        draft,
        tokensUsed: completion.usage?.total_tokens ?? 0,
        model: completion.model,
      };
    }

    function buildSystemPrompt(request: GenerationRequest): string {
      const platformContext = request.platform
        ? `You are writing content optimized for ${request.platform}.`
        : 'You are writing developer-focused content.';

      const knowledgeContext =
        request.knowledgeContext && request.knowledgeContext.length > 0
          ? `\n\nUse this context from vai documentation:\n${request.knowledgeContext.join('\n\n')}`
          : '';

      return `You are an expert developer advocate creating content for vai (voyageai-cli), a CLI tool for Voyage AI embeddings and retrieval. ${platformContext}${knowledgeContext}

Generate high-quality, technically accurate content that helps developers understand how vai solves embedding and retrieval problems.`;
    }

    function buildUserPrompt(request: GenerationRequest): string {
      const typeInstructions: Record<string, string> = {
        'blog-post': 'Write a detailed blog post (800-1500 words) with clear headings, code examples where appropriate, and a practical conclusion.',
        'social-post': 'Write a concise, engaging social post (max 300 words) with a hook, key insight, and call to action.',
        'code-example': 'Write a working, well-commented code example demonstrating the concept. Include setup instructions and expected output.',
        'video-script': 'Write a video script with intro, main content sections with timing notes, and outro. Format: [TIME] Speaker: content',
      };

      const instructions = typeInstructions[request.contentType] || 'Write engaging developer content.';
      const additional = request.additionalInstructions
        ? `\n\nAdditional instructions: ${request.additionalInstructions}`
        : '';

      return `Topic: ${request.topic}\n\n${instructions}${additional}`;
    }
    ```

    ```typescript
    import { getOpenAIClient, _resetClientForTesting } from './openai';

    describe('OpenAI client', () => {
      const originalEnv = process.env;

      beforeEach(() => {
        jest.resetModules();
        process.env = { ...originalEnv };
        _resetClientForTesting();
      });

      afterEach(() => {
        process.env = originalEnv;
        _resetClientForTesting();
      });

      it('throws when OPENAI_API_KEY is not set', () => {
        delete process.env.OPENAI_API_KEY;
        expect(() => getOpenAIClient()).toThrow('OPENAI_API_KEY environment variable is not set');
      });

      it('returns an OpenAI instance when OPENAI_API_KEY is set', () => {
        process.env.OPENAI_API_KEY = 'test-key-12345';
        const client = getOpenAIClient();
        expect(client).toBeDefined();
        expect(typeof client.chat.completions.create).toBe('function');
      });

      it('returns the same instance on repeated calls (singleton)', () => {
        process.env.OPENAI_API_KEY = 'test-key-12345';
        const client1 = getOpenAIClient();
        const client2 = getOpenAIClient();
        expect(client1).toBe(client2);
      });
    });
    ```
